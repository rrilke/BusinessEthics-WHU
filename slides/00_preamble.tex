\newcommand{\twocolslide}[5]{%
  \begin{frame}
    \frametitle{#1}
    \begin{columns}[T,onlytextwidth]
      \begin{column}{#4\textwidth} % Left column width
         \vfill
        \begin{minipage}[c][0.9\textheight][c]{\linewidth}
          #2
        \end{minipage}
        \vfill        
      \end{column}
      \begin{column}{#5\textwidth} % Right column width
       \vfill
        \begin{minipage}[c][0.9\textheight][c]{\linewidth}
          #3
        \end{minipage}
        \vfill
      \end{column}
    \end{columns}
  \end{frame}
}
